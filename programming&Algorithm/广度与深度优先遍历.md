Depth First Search - 深度优先
```js
// 深度优先-递归版本
function deepFirstSearchA(node, nodeList=[]) {

    if (node) {

        nodeList.push(node);

        const children = node.children;

        if (children) {
            for (let i = 0; i < children.length; i++)

            //每次递归的时候将 需要遍历的节点 和 节点所存储的数组传下去

            deepFirstSearch(children[i], nodeList);   
        }

    }   

    return nodeList; 
}

deepFirstSearchA(data);

// 深度优先-非递归版本
function deepFirstSearchB(node) {

    const nodes = [];

    if (node != null) {

        const stack = [];

        stack.push(node);

        while (stack.length !== 0) {

            const item = stack.pop();

            nodes.push(item);

            const children = item.children;

            if (children) {
                for (let i = children.length - 1; i >= 0; i--)
                    stack.push(children[i]);
            }
        }

    }

    return nodes;
}

deepFirstSearchB(data);
```


Breadth First Search - 广度优先

```js
// 广度优先-递归版本
function breadthFirstSearchA(stack, count=0) {
    if (stack && !Array.isArray(stack)) stack = [stack];

    if (stack[count]) {

        const {children} = stack[count];
        
        if (children) {
            for (let i = 0; i < children.length; i++)
                stack.push(children[i]);
        }

        breadthFirstSearchA(stack, ++count);
    }

    return stack;

}

breadthFirstSearchA(data);

// 广度优先-非递归版本
function breadthFirstSearchB(node) {

    const stack = [node];
    const nodes = [];

    while (stack.length) {
        const _node = stack.shift();

        nodes.push(_node);
        
        if (_node.children)
            for (let i=0;i<_node.children.length;i++)
                stack.push(_node.children[i]);
    }

    return nodes;
}

breadthFirstSearchB(data);
```
